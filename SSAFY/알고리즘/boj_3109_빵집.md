```java
package day0818;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {
	static int R,C,count,start,cur;
	static boolean[][] visited;
	static boolean flag;
	static char[][] board;
	static int[][] moves = {{-1,1},{0,1},{1,1}};
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine());

		R = Integer.parseInt(st.nextToken());
		C = Integer.parseInt(st.nextToken());

		board = new char[R][];
		visited = new boolean[R][C];
		for(int i = 0; i < R; i++) {
			board[i] = br.readLine().toCharArray();
		}
		
		for(int i = 0; i < R; i++) {
			if(solve(i,0)) count+=1;
		}

		System.out.println(count);
	}

	private static boolean solve(int x, int y) {		
		if(y == C-1){
			return true;
		}
		visited[x][y] = true;
		
		for(int i = 0; i < 3; i++) {
			int nx = x + moves[i][0];
			int ny = y + moves[i][1];
			if((nx < 0 || nx >= R || ny < 0 || ny >= C) || board[nx][ny] != '.' || visited[nx][ny]) continue;
			
			board[nx][ny] = '-';
			visited[nx][ny] = true;
			
			if(solve(nx,ny)) return true;
			board[nx][ny] = '.';
		}
		return false;
	}
}

```

- 결국 board[][]를 기준으로 가장 위쪽으로 배치될 수 있는 파이프의 경우를 먼저 세는 것이 최대값을 도출하기에 유리하기 떄문에 해당 방법을 채택하여 풀이하는 것이다.
