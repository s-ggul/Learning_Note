```java
package com.saffy.recur;

public class Test01 {

	public static void main(String[] args) {
		System.out.println("반복을 이용한 합계 : "+ solve1(10));
		System.out.println("반환형O(10 => 0)재귀를 이용한 합계 : " + solve2(10));
		System.out.println("반환형O(0 => 10)재귀를 이용한 합계 : " + solve4(0,0));
		solve3(10);
		System.out.println("void재귀를 이용한 합계 : " + sum);
		System.out.println("Tail Recursive 를 이용한 합계 : " + solve5(10, 0));
	}
	
	public static int solve1(int num){
		int sum = 0;
		
		for(int i = 0; i <= num; i++) {
			sum += i;
		}
		
		return sum;
	}
	
	// 10 => 0
	public static int solve2(int num) {
		if(num == 0) return num;
		return num + solve2(num-1);
	}
	
	// 10 => 0
	static int sum = 0;
	public static void solve3(int num) {
		if(num == 0) return;
		
		solve3(num - 1);
		sum += num;
	}
	
	// 0 => 10
	public static int solve4(int num, int count) {
		if(count == 10) {
			return num;
		}
		
		return num + solve4(num+1,count+1);
	}
	
	// tail recursion
	public static int solve5(int num, int sum) {
		if(num == 0) return sum; 
		return solve5(num-1, sum + num);
	}
}
```

```java
package com.saffy.recur;

public class Test02 {

	public static void main(String[] args) {
		System.out.println("------------------ 내 풀이 ------------------");
		System.out.println("반복을 이용한 자리수의 합 구하기 : " + solve1(13689));
		System.out.println("재귀를 이용한 자리수의 합 구하기 : " + solve2(13689,0));
		System.out.println("\n----------------- 교수님풀이 -----------------");
		System.out.println("반복을 이용한 자리수의 합 구하기 : " + solve3(13689));
		System.out.println("재귀를 이용한 자리수의 합 구하기 : " + solve4(13689));
	}

	// ------------------ 내 풀이 ------------------
	
	// while 반복문
	public static int solve1(int num) {
		int sum = 0;
		while(true) {
			if(num == 0) {
				return sum;
			}		
			sum += (num%10);
			num /= 10;
		}
	}
	
	// Tail Recursion
	public static int solve2(int num, int sum) {
		if(num == 0) return sum;
		return solve2(num/10, sum + (num%10));
	}
	
	
	// ----------------- 교수님풀이 -----------------

	// while 반복문
	public static int solve3(int num) {
		int ans = 0;
		while(true) {
			if(num == 0) return ans;
			
			ans += num % 10;
			num /= 10;
		}
	}
	
	// 재귀
	public static int solve4(int num) {
		if(num < 10) return num;
		return num % 10 + solve4(num/10);
	}
}
```

```java
package com.saffy.recur;

public class Test03 {
	public static void main(String[] args) {
		int[] arr = {1,2,3,4,5};
	
		// solve 함수구현 : 매개변수는 자유롭게 설정
		// 배열의 합 구하기 
		System.out.println(solve1(arr, 0));		
	}

	private static int solve1(int[] arr, int index) {
		if(index == arr.length) return 0;
		return arr[index] + solve1(arr, index+1);
	}
}
```

#### 교수님 과제 풀이 - 백준 1244 : 스위치 켜고 끄기

```java
package com.ssafy.boj;

import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.util.Scanner;

public class BOJ_1244_스위치켜고끄기 {

	public static void main(String[] args) throws FileNotFoundException {
		System.setIn(new FileInputStream("data/1244.txt"));
		Scanner sc = new Scanner(System.in);
		int N = sc.nextInt();
		int[] data = new int[N+1];
		for(int i = 1; i <= N; i++) {
			data[i] = sc.nextInt();
		}

		int stuNum = sc.nextInt();
		int loc = 0;
		int[] change = {1,0};
		for(int i = 0; i < stuNum; i++) {
			switch(sc.nextInt()) {
			case 1: // 남학생
				loc = sc.nextInt();
				for(int j = loc; j <= N; j += loc) {
					data[j] = change[data[j]];
					// 0일땐 1, 1일땐 0 ... 이 생각은 가지고 있는게 좋을 듯함.
				}
				break;
			case 2: // 여학생
				loc = sc.nextInt();
				int l = loc, r = loc; //투포인터
				while(l >= 1 && r <=N) {
					if(data[l] != data[r]) break;
					
					data[l] = data[r] = change[data[l]];
					l--;
					r++;
				}
				break;
			}
		}
		for(int i = 1; i <= N; i++) {
			System.out.printf("%d%s",data[i],i % 20 == 0 ? "\n":" ");
		}
		sc.close();
	}
}
```
