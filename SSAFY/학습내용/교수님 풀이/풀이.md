void call(int x) {
  x += x;
  System.out.println( x );  ->  이건 메서드에 새롭게 선언된 x 변수 : 10 출력
}

main () {
    int x = 5;
    call(x);
    System.out.println(x);  -> 이건 메인 메서드에 선언된 x 변수 : 5 출력
}
